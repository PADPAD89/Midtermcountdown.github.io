<script>
  const ELECTION = new Date('2026-11-03T06:00:00Z'); // polls open
  const TERM_START = new Date('2025-01-20T12:00:00Z');
  const TERM_END   = ELECTION;

  const $d  = document.getElementById('days');
  const $h  = document.getElementById('hours');
  const $m  = document.getElementById('mins');
  const $s  = document.getElementById('secs');
  const $p  = document.getElementById('progress');
  const $pt = document.getElementById('pct');

  function pad(n, len = 2) { return String(n).padStart(len, '0'); }

  function animateFlip(el, val) {
    const padded = el.id === 'days' ? pad(val, 3) : pad(val, 2);
    if (el.textContent !== padded) {
      el.classList.remove('flip');
      void el.offsetWidth; // reflow
      el.classList.add('flip');
      el.textContent = padded;
    }
  }

  function tick() {
    const now  = new Date();
    const diff = ELECTION - now;

    if (diff <= 0) {
      $d.textContent = '000';
      $h.textContent = '00';
      $m.textContent = '00';
      $s.textContent = '00';
      document.querySelector('.sub-title').textContent = '🗳 Election Day is HERE!';
      return;
    }

    const totalSecs = Math.floor(diff / 1000);
    const secs  = totalSecs % 60;
    const mins  = Math.floor(totalSecs / 60) % 60;
    const hours = Math.floor(totalSecs / 3600) % 24;
    const days  = Math.floor(totalSecs / 86400);

    animateFlip($d, days);
    animateFlip($h, hours);
    animateFlip($m, mins);
    animateFlip($s, secs);

    // Progress bar: % of time from inauguration → election day elapsed
    const totalSpan = TERM_END - TERM_START;
    const elapsed   = now - TERM_START;
    const pct = Math.min(100, Math.max(0, (elapsed / totalSpan) * 100));
    $p.style.width = pct.toFixed(2) + '%';
    $pt.textContent = pct.toFixed(1) + '% elapsed since Jan 20, 2025';
  }

  tick();
  setInterval(tick, 1000);
</script>
